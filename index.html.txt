<script>
const titleEl = document.getElementById("title");
const textEl = document.getElementById("text");
const btnEl = document.getElementById("buttons");

/* NAME TYPING */
function typeTitle(name, callback){
    let i = 0;
    function type(){
        if(i < name.length){
            titleEl.innerHTML += name.charAt(i);
            i++;
            setTimeout(type, 120);
        } else {
            setTimeout(callback, 500);
        }
    }
    type();
}

/* TEXT TYPING */
function typeText(message, callback){
    textEl.innerHTML = "";
    btnEl.innerHTML = "";
    let i = 0;

    function type(){
        if(i < message.length){
            textEl.innerHTML += message.charAt(i);
            i++;
            setTimeout(type, 25);
        } else if(callback){
            callback();
        }
    }
    type();
}

/* START */
function start(){
    typeTitle("Hey Nikki… 💗", () => {
        typeText(
`Before you click anything…

This is not a question…
and there is nothing you need to answer…

I made this because you are someone special to me 🌸`,
        ()=> btnEl.innerHTML = `<button onclick="next1()">Continue 💙</button>`);
    });
}

function next1(){
    typeText(
`So just sit back…
and read this slowly 😊`,
    ()=> btnEl.innerHTML = `<button onclick="feeling()">Continue 🌙</button>`);
}

`How are you feeling right now? 🌙

(Just be honest… no pressure 💛)`,
    ()=>{
        btnEl.innerHTML = `
        <button onclick="happy()">😊 Happy</button>
        <button onclick="sad()">😔 Sad</button>
        <button onclick="confused()">😶 Confused</button>`;
    });
}

function happy(){ typeText("I’m really glad 😊… stay like this always 💛", nextCommon); }
